2.)SELECT * FROM Worker ORDER BY FIRST_NAME ASC, DEPARTMENT DESC;

3.)SELECT * FROM Worker WHERE FIRST_NAME NOT IN ('Vipul', 'Satish');

4.)SELECT * FROM Worker WHERE FIRST_NAME LIKE '_____h';

5.)SELECT COUNT(*) AS Admin_Employee_Count FROM Worker WHERE DEPARTMENT = 'Admin';

6.)SELECT FIRST_NAME, LAST_NAME, SALARY FROM Worker WHERE SALARY BETWEEN 50000 AND 100000;

7.)SELECT DEPARTMENT, COUNT(*) AS Worker_Count FROM Worker GROUP BY DEPARTMENT ORDER BY Worker_Count DESC;

8.)SELECT DISTINCT SALARY FROM Worker W1 WHERE 4 = (
    SELECT COUNT(DISTINCT SALARY) 
    FROM Worker W2 
    WHERE W2.SALARY > W1.SALARY
);

9.)SELECT W1.FIRST_NAME, W1.LAST_NAME, W1.SALARY FROM Worker W1
WHERE W1.SALARY IN (
    SELECT SALARY 
    FROM Worker 
    GROUP BY SALARY 
    HAVING COUNT(*) > 1
);

10.)SELECT DEPARTMENT FROM Worker GROUP BY DEPARTMENT HAVING COUNT(*) < 3;
